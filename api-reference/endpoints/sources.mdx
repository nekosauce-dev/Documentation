---
title: "Get Sources"
api: "GET https://nekosauce.org/api/sources"
description: "This endpoint returns all available sources."
---

### Response

<ResponseField name="data" type="object[]">
  Contains all the results.

  <Expandable title="data">
    <ResponseField name="id" type="number">
      The source's ID.
    </ResponseField>
    <ResponseField name="module" type="string">
      The internal import module. Not useful for API clients but it's there for internal use.
    </ResponseField>
    <ResponseField name="name" type="string">
      The human-readable name of the source.
    </ResponseField>
    <ResponseField name="urls" type="object">
      The source's URLs.

      <Expandable title="urls">
        <ResponseField name="website" type="string">
          The source's official website URL.
        </ResponseField>
        <ResponseField name="documentation" type="string">
          The source's official API documentation.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="enabled" type="boolean">
      Whether the source is enabled or not. If `false`, the source won't be updated with new sauces.
    </ResponseField>
    <ResponseField name="components" type="object">
      The internal components, located at `module`. They're generally useless for API clients.

      <Expandable name="components">
        <ResponseField name="fetcher" type="string">
          The class name of the source's fetcher. Instances of this class are the ones that fetch new sauces and add them to the database. If it's `null`, this source doesn't fetch sauces.
        </ResponseField>
        <ResponseField name="downloader" type="string">
          The class name of the source's downloader. Instances of this class are the ones that download the images from the source. If it's `null`, this source doesn't download images.
        </ResponseField>
        <ResponseField name="tagger" type="string">
          The class name of the source's tagger. Instances of this class are the ones that convert sauce URLs into tags like `danbooru:post:id:1`.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="sauce_types" type="number[]">
      The sauce types this source can have.
    </ResponseField>
    <ResponseField name="update_frequency" type="string">
      The frequency with which new sauces are fetched from the source.
    </ResponseField>
  </Expandable>
</ResponseField>
